import error_handling


def input_sss():

    try:
        s1 = float(input("Enter side 1: "))
        s2 = float(input("Enter side 2: "))
        s3 = float(input("Enter side 3: "))
    except Exception as e:
        raise Exception(e)
    else:
        try:
            error_handling.validate_input("side", s1, s2, s3)
        except Exception as e:
            raise Exception(e)

    sort = sorted([s1, s2, s3])
    return (
        sort[0],
        sort[1],
        sort[2],
    )
